#install miniconda
1. Open Powershell
2. winget install -e --id Anaconda.Miniconda3
3. Open anaconda powershell prompt (miniconda3) from windows start menu 
4. cd "Directory of yolov7"
5. conda env create -f environment.yml
6. conda activate yolov7
7. python detect.py --weights best.pt --conf 0.5 --source "Location of source folder/file"
